```sql
-- Opaque, short-lived challenge issued after password verification when the

  on auth_mfa_challenges (user_id);

  on auth_mfa_challenges (expires_at)

-- +goose StatementBegin
```

**`backend/internal/auth/challenge.go`:**

```go
package auth

import (
	"context"
	"encoding/json"
	"errors"
	"net"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgxpool"
)

// MFAChallenge is an ephemeral post-password ticket. The client presents the
// challenge ID to `POST /auth/login/mfa/*` to complete the second factor.
type MFAChallenge struct {
	ID            uuid.UUID
	UserID        uuid.UUID
	IP            net.IP
	UserAgent     string
	CreatedAt     time.Time
	ExpiresAt     time.Time
	ConsumedAt    *time.Time
	Attempts      int
	WebAuthnState []byte // opaque webauthn session JSON
}

var (
	ErrChallengeNotFound = errors.New("mfa challenge not found")
	ErrChallengeExpired  = errors.New("mfa challenge expired")
	ErrChallengeBinding  = errors.New("mfa challenge ip/ua mismatch")
	ErrChallengeConsumed = errors.New("mfa challenge already consumed")
)

// InsertMFAChallenge creates a challenge row. Caller must already have
// verified the password. IP/UA bind the challenge to the original request.
func InsertMFAChallenge(ctx context.Context, db *pgxpool.Pool, c MFAChallenge) error {
	_, err := db.Exec(ctx, `
		insert into auth_mfa_challenges
			(id, user_id, ip, user_agent, created_at, expires_at, webauthn_state)
		values ($1, $2, $3, $4, $5, $6, $7)
	`, c.ID, c.UserID, c.IP.String(), c.UserAgent, c.CreatedAt, c.ExpiresAt, c.WebAuthnState)
	return err
}

// LoadAndBindMFAChallenge returns the row if it exists, is live, and the
// caller's IP + UA match. The caller passes `now` so the fake clock threads
// through in tests.
func LoadAndBindMFAChallenge(ctx context.Context, db *pgxpool.Pool, id uuid.UUID, ip net.IP, ua string, now time.Time) (MFAChallenge, error) {
	var c MFAChallenge
	var ipStr string
	err := db.QueryRow(ctx, `
		select id, user_id, ip::text, user_agent, created_at, expires_at, consumed_at, attempts, webauthn_state
		from auth_mfa_challenges
		where id = $1
	`, id).Scan(&c.ID, &c.UserID, &ipStr, &c.UserAgent, &c.CreatedAt, &c.ExpiresAt, &c.ConsumedAt, &c.Attempts, &c.WebAuthnState)
	if errors.Is(err, pgx.ErrNoRows) {
		return MFAChallenge{}, ErrChallengeNotFound
	}
	if err != nil {
		return MFAChallenge{}, err
	}
	c.IP = net.ParseIP(ipStr)
	if c.ConsumedAt != nil {
		return c, ErrChallengeConsumed
	}
	if now.After(c.ExpiresAt) {
		return c, ErrChallengeExpired
	}
	if !c.IP.Equal(ip) || c.UserAgent != ua {
		return c, ErrChallengeBinding
	}
	return c, nil
}

// ConsumeMFAChallenge marks the challenge consumed in the same tx that
// issues the session. Returns ErrChallengeConsumed if already consumed.
func ConsumeMFAChallenge(ctx context.Context, tx pgx.Tx, id uuid.UUID, now time.Time) error {
	ct, err := tx.Exec(ctx, `
		update auth_mfa_challenges
		set consumed_at = $2
		where id = $1 and consumed_at is null
	`, id, now)
	if err != nil {
		return err
	}
	if ct.RowsAffected() == 0 {
		return ErrChallengeConsumed
	}
	return nil
}

// BumpAttempts increments attempts; >=10 invalidates the challenge (§8.2).
func BumpAttempts(ctx context.Context, db *pgxpool.Pool, id uuid.UUID, now time.Time) (int, error) {
	var attempts int
	err := db.QueryRow(ctx, `
		update auth_mfa_challenges
		set attempts = attempts + 1,
		    consumed_at = case when attempts + 1 >= 10 then $2 else consumed_at end
		where id = $1
		returning attempts
	`, id, now).Scan(&attempts)
	return attempts, err
}

// UpdateWebAuthnState persists the webauthn library's session data between
// Begin and Complete calls. Encodes via json.Marshal for stability.
func UpdateWebAuthnState(ctx context.Context, db *pgxpool.Pool, id uuid.UUID, state any) error {
	b, err := json.Marshal(state)
	if err != nil {
		return err
	}
	_, err = db.Exec(ctx, `update auth_mfa_challenges set webauthn_state = $2 where id = $1`, id, b)
	return err
}
```

```go
	userID := mustUser(t, svc, "alice@example.com")
	now := time.Date(2026, 4, 24, 12, 0, 0, 0, time.UTC)

	c := auth.MFAChallenge{
		ID:        uuid.Must(uuid.NewV7()),
		UserID:    userID,
		IP:        net.ParseIP("192.0.2.5"),
		UserAgent: "ua-test",
		CreatedAt: now,

	// Good load
	got, err := auth.LoadAndBindMFAChallenge(ctx, pool, c.ID, c.IP, c.UserAgent, now)
	require.NoError(t, err)

	require.NoError(t, auth.ConsumeMFAChallenge(ctx, tx, c.ID, now))
	require.NoError(t, tx.Commit(ctx))
```

```go
package auth

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"errors"
	"fmt"
	"io"
)

// SealSecret AES-GCM-encrypts plaintext using key (must be 32 bytes).
// The output layout is nonce || ciphertext || tag; decoded by OpenSecret.
func SealSecret(key, plaintext []byte) ([]byte, error) {
	if len(key) != 32 {
		return nil, errors.New("secret key must be 32 bytes")
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, err
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, err
	}
	nonce := make([]byte, gcm.NonceSize())
	if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
		return nil, err
	}
	return gcm.Seal(nonce, nonce, plaintext, nil), nil
}

// OpenSecret reverses SealSecret. Returns a clear error on tamper.
func OpenSecret(key, sealed []byte) ([]byte, error) {
	if len(key) != 32 {
		return nil, errors.New("secret key must be 32 bytes")
	}
	block, err := aes.NewCipher(key)
	if err != nil {
		return nil, err
	}
	gcm, err := cipher.NewGCM(block)
	if err != nil {
		return nil, err
	}
	if len(sealed) < gcm.NonceSize() {
		return nil, fmt.Errorf("sealed too short")
	}
	nonce := sealed[:gcm.NonceSize()]
	ct := sealed[gcm.NonceSize():]
	return gcm.Open(nil, nonce, ct, nil)
}
```

**`backend/internal/auth/totp.go`:**

```go
package auth

import (
	"bytes"
	"context"
	"encoding/base64"
	"errors"
	"fmt"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
	"github.com/pquerna/otp/totp"
	"github.com/skip2/go-qrcode"
)

// TOTPSetup is returned from EnrollTOTP. Secret is shown once so the user
// can store it in their password manager; QRCodeBase64 is a data-URI-ready
// PNG. RecoveryCodes are empty here — they're generated on ConfirmTOTP.
type TOTPSetup struct {
	Secret        string
	URI           string
	QRCodeBase64  string
	RecoveryCodes []string
}

var (
	ErrTOTPAlreadyEnrolled = errors.New("totp already enrolled")
	ErrTOTPNotEnrolled     = errors.New("totp not enrolled")
	ErrTOTPInvalidCode     = errors.New("invalid totp code")
)

// EnrollTOTP generates a new TOTP secret, stores it unverified, and returns
// the provisioning URI + QR PNG for display. verified_at stays null until
// ConfirmTOTP succeeds.
func (s *Service) EnrollTOTP(ctx context.Context, userID uuid.UUID) (TOTPSetup, error) {
	user, err := s.GetUserByID(ctx, userID)
	if err != nil {
		return TOTPSetup{}, err
	}

	// Fail fast if already enrolled+verified.
	var existingVerified *time.Time
	err = s.db.QueryRow(ctx, `
		select verified_at from totp_credentials where user_id = $1
	`, userID).Scan(&existingVerified)
	if err == nil && existingVerified != nil {
		return TOTPSetup{}, ErrTOTPAlreadyEnrolled
	}

	key, err := totp.Generate(totp.GenerateOpts{
		Issuer:      "Folio",
		AccountName: user.Email,
		Period:      30,
		Digits:      6,
	})
	if err != nil {
		return TOTPSetup{}, err
	}

	cipher, err := SealSecret(s.cfg.SecretEncryptionKey, []byte(key.Secret()))
	if err != nil {
		return TOTPSetup{}, err
	}

	// Upsert — replaces any prior unverified attempt.
	_, err = s.db.Exec(ctx, `
		insert into totp_credentials (user_id, secret_cipher, created_at)
		values ($1, $2, $3)
		on conflict (user_id) do update
		set secret_cipher = excluded.secret_cipher,
		    created_at    = excluded.created_at,
		    verified_at   = null
	`, userID, cipher, s.clock.Now())
	if err != nil {
		return TOTPSetup{}, err
	}

	png, err := qrcode.Encode(key.URL(), qrcode.Medium, 256)
	if err != nil {
		return TOTPSetup{}, err
	}

	return TOTPSetup{
		Secret:       key.Secret(),
		URI:          key.URL(),
		QRCodeBase64: base64.StdEncoding.EncodeToString(png),
	}, nil
}

// ConfirmTOTP validates the first code, flips verified_at, and returns a
// freshly-generated set of recovery codes (replaces any prior).
func (s *Service) ConfirmTOTP(ctx context.Context, userID uuid.UUID, code string) ([]string, error) {
	secret, err := s.loadTOTPSecret(ctx, userID)
	if err != nil {
		return nil, err
	}
	if !totp.Validate(code, secret) {
		return nil, ErrTOTPInvalidCode
	}

	tx, err := s.db.Begin(ctx)
	if err != nil {
		return nil, err
	}
	defer tx.Rollback(ctx)

	now := s.clock.Now()
	if _, err := tx.Exec(ctx,
		`update totp_credentials set verified_at = $2 where user_id = $1`,
		userID, now,
	); err != nil {
		return nil, err
	}

	codes, err := s.generateAndStoreRecoveryCodes(ctx, tx, userID, now)
	if err != nil {
		return nil, err
	}

	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.totp_enabled",
		EntityType:  "user",
		EntityID:    userID.String(),
		At:          now,
	})

	return codes, tx.Commit(ctx)
}

// DisableTOTP is step-up-gated at the handler; service just deletes.
func (s *Service) DisableTOTP(ctx context.Context, userID uuid.UUID) error {
	tx, err := s.db.Begin(ctx)
	if err != nil {
		return err
	}
	defer tx.Rollback(ctx)

	ct, err := tx.Exec(ctx, `delete from totp_credentials where user_id = $1`, userID)
	if err != nil {
		return err
	}
	if ct.RowsAffected() == 0 {
		return ErrTOTPNotEnrolled
	}
	// Revoke all other sessions per §6.3 (credential removed).
	if err := s.revokeAllSessionsExceptCurrentTx(ctx, tx, userID); err != nil {
		return err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.totp_disabled",
		EntityType:  "user",
		EntityID:    userID.String(),
		At:          s.clock.Now(),
	})
	return tx.Commit(ctx)
}

// verifyTOTPCode validates a ±1-period window (30s drift either way).
func (s *Service) verifyTOTPCode(ctx context.Context, userID uuid.UUID, code string) error {
	secret, err := s.loadTOTPSecret(ctx, userID)
	if err != nil {
		return err
	}
	ok, err := totp.ValidateCustom(code, secret, s.clock.Now(), totp.ValidateOpts{
		Period:    30,
		Skew:      1,
		Digits:    6,
		Algorithm: 0, // SHA1 per RFC 6238 default
	})
	if err != nil {
		return err
	}
	if !ok {
		return ErrTOTPInvalidCode
	}
	return nil
}

func (s *Service) loadTOTPSecret(ctx context.Context, userID uuid.UUID) (string, error) {
	var cipher []byte
	err := s.db.QueryRow(ctx, `
		select secret_cipher from totp_credentials where user_id = $1
	`, userID).Scan(&cipher)
	if errors.Is(err, pgx.ErrNoRows) {
		return "", ErrTOTPNotEnrolled
	}
	if err != nil {
		return "", err
	}
	plain, err := OpenSecret(s.cfg.SecretEncryptionKey, cipher)
	if err != nil {
		return "", fmt.Errorf("decrypt totp secret: %w", err)
	}
	return string(bytes.TrimSpace(plain)), nil
```

```go
package auth

import (
	"context"
	"crypto/rand"
	"encoding/base32"
	"errors"
	"strings"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"

	"folio/backend/internal/argon2id"
)

var ErrRecoveryCodeInvalid = errors.New("recovery code invalid")

const recoveryCodeCount = 10

// generateRecoveryCode returns a 10-char base32 string (crockford-style
// uppercase). Format shown to user: XXXXX-XXXXX for readability.
func generateRecoveryCode() (string, error) {
	buf := make([]byte, 10)
	if _, err := rand.Read(buf); err != nil {
		return "", err
	}
	enc := base32.StdEncoding.WithPadding(base32.NoPadding).EncodeToString(buf)
	enc = strings.ToUpper(enc)[:10]
	return enc[:5] + "-" + enc[5:], nil
}

// normalize strips dashes + spaces and uppercases so "abcde-fghij" matches
// "ABCDEFGHIJ".
func normalizeRecoveryCode(s string) string {
	return strings.ToUpper(strings.NewReplacer("-", "", " ", "").Replace(s))
}

// generateAndStoreRecoveryCodes deletes previous codes and inserts 10 new
// rows, each with an Argon2id hash of the normalized plaintext.
func (s *Service) generateAndStoreRecoveryCodes(ctx context.Context, tx pgx.Tx, userID uuid.UUID, now time.Time) ([]string, error) {
	if _, err := tx.Exec(ctx, `delete from auth_recovery_codes where user_id = $1`, userID); err != nil {
		return nil, err
	}
	plain := make([]string, 0, recoveryCodeCount)
	for i := 0; i < recoveryCodeCount; i++ {
		code, err := generateRecoveryCode()
		if err != nil {
			return nil, err
		}
		hash, err := argon2id.Hash(normalizeRecoveryCode(code))
		if err != nil {
			return nil, err
		}
		if _, err := tx.Exec(ctx, `
			insert into auth_recovery_codes (id, user_id, code_hash, created_at)
			values ($1, $2, $3, $4)
		`, uuid.Must(uuid.NewV7()), userID, hash, now); err != nil {
			return nil, err
		}
		plain = append(plain, code)
	}
	return plain, nil
}

// consumeRecoveryCode finds a live row whose Argon2id hash matches, marks it
// consumed, and returns nil on success.
func (s *Service) consumeRecoveryCode(ctx context.Context, tx pgx.Tx, userID uuid.UUID, code string, now time.Time) error {
	rows, err := tx.Query(ctx, `
		select id, code_hash from auth_recovery_codes
		where user_id = $1 and consumed_at is null
		for update
	`, userID)
	if err != nil {
		return err
	}
	defer rows.Close()

	normalized := normalizeRecoveryCode(code)
	var matchID *uuid.UUID
	for rows.Next() {
		var id uuid.UUID
		var hash string
		if err := rows.Scan(&id, &hash); err != nil {
			return err
		}
		ok, _ := argon2id.Verify(normalized, hash)
		if ok {
			matchID = &id
			break
		}
	}
	rows.Close()
	if matchID == nil {
		return ErrRecoveryCodeInvalid
	}
	_, err = tx.Exec(ctx, `
		update auth_recovery_codes set consumed_at = $2 where id = $1
	`, *matchID, now)
	return err
}

// RegenerateRecoveryCodes replaces all codes for a user. Caller must have
// passed step-up at the HTTP layer.
func (s *Service) RegenerateRecoveryCodes(ctx context.Context, userID uuid.UUID) ([]string, error) {
	tx, err := s.db.Begin(ctx)
	if err != nil {
		return nil, err
	}
	defer tx.Rollback(ctx)

	now := s.clock.Now()
	codes, err := s.generateAndStoreRecoveryCodes(ctx, tx, userID, now)
	if err != nil {
		return nil, err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.recovery_codes_regenerated",
		EntityType:  "user",
		EntityID:    userID.String(),
		At:          now,
	})
	return codes, tx.Commit(ctx)
}

// RemainingRecoveryCodes is surfaced on `GET /me/mfa`.
func (s *Service) RemainingRecoveryCodes(ctx context.Context, userID uuid.UUID) (int, error) {
	var n int
	err := s.db.QueryRow(ctx, `
		select count(*) from auth_recovery_codes
		where user_id = $1 and consumed_at is null
	`, userID).Scan(&n)
	return n, err
}
```

```go
// WebAuthnClient is a thin wrapper around webauthn.WebAuthn so the service
// layer stays testable.
type WebAuthnClient struct {
	W *webauthn.WebAuthn
}

// PasskeyCredential is the Go row for webauthn_credentials.
type PasskeyCredential struct {
	ID              uuid.UUID
	UserID          uuid.UUID
	CredentialID    []byte
	PublicKey       []byte
	AAGUID          []byte
	SignCount       uint32
	Label           string
	Transports      []string
	CreatedAt       time.Time
	LastUsedAt      *time.Time
}

// webauthnUser adapts our user+credentials to the webauthn library.
type webauthnUser struct {
	id          []byte
	name        string
	displayName string
	creds       []webauthn.Credential
}

func (u *webauthnUser) WebAuthnID() []byte                         { return u.id }
func (u *webauthnUser) WebAuthnName() string                       { return u.name }
func (u *webauthnUser) WebAuthnDisplayName() string                { return u.displayName }
func (u *webauthnUser) WebAuthnCredentials() []webauthn.Credential { return u.creds }
func (u *webauthnUser) WebAuthnIcon() string                       { return "" }

// loadWebAuthnUser hydrates a user + their stored credentials for assertion.
func (s *Service) loadWebAuthnUser(ctx context.Context, db *pgxpool.Pool, userID uuid.UUID) (*webauthnUser, error) {
	u, err := s.GetUserByID(ctx, userID)
	if err != nil {
		return nil, err
	}
	rows, err := db.Query(ctx, `
		select credential_id, public_key, aaguid, sign_count, transports
		from webauthn_credentials where user_id = $1
	`, userID)
	if err != nil {
		return nil, err
	}
	defer rows.Close()

	var creds []webauthn.Credential
	for rows.Next() {
		var c webauthn.Credential
		var transports []string
		if err := rows.Scan(&c.ID, &c.PublicKey, &c.Authenticator.AAGUID, &c.Authenticator.SignCount, &transports); err != nil {
			return nil, err
		}
		for _, t := range transports {
			c.Transport = append(c.Transport, protocol.AuthenticatorTransport(t))
		}
		creds = append(creds, c)
	}

	return &webauthnUser{
		id:          u.ID[:],
		name:        u.Email,
		displayName: u.DisplayName,
		creds:       creds,
	}, nil
}

// EnrollPasskey kicks off registration; returns the CredentialCreation options
// to send to the browser. Session data is stored in-memory on Service.webauthnSessions
// keyed by userID for single-device dev; prod routes through auth_mfa_challenges.
func (s *Service) EnrollPasskey(ctx context.Context, userID uuid.UUID) (*protocol.CredentialCreation, error) {
	wu, err := s.loadWebAuthnUser(ctx, s.db, userID)
	if err != nil {
		return nil, err
	}
	options, sessionData, err := s.webauthn.W.BeginRegistration(wu,
		webauthn.WithResidentKeyRequirement(protocol.ResidentKeyRequirementPreferred),
		webauthn.WithAuthenticatorSelection(protocol.AuthenticatorSelection{
			UserVerification:   protocol.VerificationPreferred,
			ResidentKey:        protocol.ResidentKeyRequirementPreferred,
		}),
	)
	if err != nil {
		return nil, err
	}
	s.saveWebAuthnSession(userID, sessionData)
	return options, nil
}

// CompleteEnrollPasskey parses the browser's attestation response, validates
// it, and inserts a row in `webauthn_credentials`.
func (s *Service) CompleteEnrollPasskey(ctx context.Context, userID uuid.UUID, response []byte, label string) (PasskeyCredential, error) {
	wu, err := s.loadWebAuthnUser(ctx, s.db, userID)
	if err != nil {
		return PasskeyCredential{}, err
	}
	sessionData, ok := s.loadWebAuthnSession(userID)
	if !ok {
		return PasskeyCredential{}, errors.New("no registration session in progress")
	}
	parsed, err := protocol.ParseCredentialCreationResponseBody(base64.NewDecoder(base64.StdEncoding, bytesReader(response)))
	if err != nil {
		return PasskeyCredential{}, err
	}
	cred, err := s.webauthn.W.CreateCredential(wu, sessionData, parsed)
	if err != nil {
		return PasskeyCredential{}, err
	}
	s.deleteWebAuthnSession(userID)

	id := uuid.Must(uuid.NewV7())
	now := s.clock.Now()
	transports := make([]string, 0, len(cred.Transport))
	for _, t := range cred.Transport {
		transports = append(transports, string(t))
	}

	tx, err := s.db.Begin(ctx)
	if err != nil {
		return PasskeyCredential{}, err
	}
	defer tx.Rollback(ctx)

	if _, err := tx.Exec(ctx, `
		insert into webauthn_credentials
			(id, user_id, credential_id, public_key, aaguid, sign_count, label, transports, created_at)
		values ($1, $2, $3, $4, $5, $6, $7, $8, $9)
	`, id, userID, cred.ID, cred.PublicKey, cred.Authenticator.AAGUID, cred.Authenticator.SignCount, label, transports, now); err != nil {
		return PasskeyCredential{}, err
	}

	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.passkey_added",
		EntityType:  "webauthn_credential",
		EntityID:    id.String(),
		At:          now,
	})

	if err := tx.Commit(ctx); err != nil {
		return PasskeyCredential{}, err
	}

	return PasskeyCredential{
		ID:           id,
		UserID:       userID,
		CredentialID: cred.ID,
		PublicKey:    cred.PublicKey,
		AAGUID:       cred.Authenticator.AAGUID,
		SignCount:    cred.Authenticator.SignCount,
		Label:        label,
		Transports:   transports,
		CreatedAt:    now,
	}, nil
}

// DeletePasskey removes a credential by its UUID (not credential_id). Step-up
// is enforced at the HTTP layer.
func (s *Service) DeletePasskey(ctx context.Context, userID, credentialID uuid.UUID) error {
	tx, err := s.db.Begin(ctx)
	if err != nil {
		return err
	}
	defer tx.Rollback(ctx)

	ct, err := tx.Exec(ctx, `
		delete from webauthn_credentials where id = $1 and user_id = $2
	`, credentialID, userID)
	if err != nil {
		return err
	}
	if ct.RowsAffected() == 0 {
		return errors.New("passkey not found")
	}
	// Credential removed → revoke other sessions per §6.3.
	if err := s.revokeAllSessionsExceptCurrentTx(ctx, tx, userID); err != nil {
		return err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.passkey_removed",
		EntityType:  "webauthn_credential",
		EntityID:    credentialID.String(),
		At:          s.clock.Now(),
	})
	return tx.Commit(ctx)
}

		}

	}
	return out, nil
}
```

```go
// BeginWebAuthnAssertion starts a passkey assertion for a user who already
// completed the password step. Returns a CredentialAssertion payload for the
// browser; the webauthn session state is persisted on the challenge row.
func (s *Service) BeginWebAuthnAssertion(ctx context.Context, challengeID uuid.UUID, ip, ua string) (*protocol.CredentialAssertion, error) {
	challenge, err := LoadAndBindMFAChallenge(ctx, s.db, challengeID, parseIP(ip), ua, s.clock.Now())
	if err != nil {
		return nil, err
	}
	wu, err := s.loadWebAuthnUser(ctx, s.db, challenge.UserID)
	if err != nil {
		return nil, err
	}
	if len(wu.creds) == 0 {
		return nil, errors.New("user has no passkeys enrolled")
	}
	options, sessionData, err := s.webauthn.W.BeginLogin(wu)
	if err != nil {
		return nil, err
	}
	if err := UpdateWebAuthnState(ctx, s.db, challengeID, sessionData); err != nil {
		return nil, err
	}
	return options, nil
}

// CompleteWebAuthnAssertion finishes the ceremony, consumes the challenge,
// and issues a session.
func (s *Service) CompleteWebAuthnAssertion(ctx context.Context, challengeID uuid.UUID, response []byte, ip, ua string) (string, error) {
	now := s.clock.Now()
	challenge, err := LoadAndBindMFAChallenge(ctx, s.db, challengeID, parseIP(ip), ua, now)
	if err != nil {
		return "", err
	}

	wu, err := s.loadWebAuthnUser(ctx, s.db, challenge.UserID)
	if err != nil {
		return "", err
	}

	var sessionData webauthn.SessionData
	if err := json.Unmarshal(challenge.WebAuthnState, &sessionData); err != nil {
		return "", err
	}

	parsed, err := protocol.ParseCredentialRequestResponseBody(bytes.NewReader(response))
	if err != nil {
		return "", err
	}
	cred, err := s.webauthn.W.ValidateLogin(wu, sessionData, parsed)
	if err != nil {
		if _, bumpErr := BumpAttempts(ctx, s.db, challengeID, now); bumpErr != nil {
			return "", bumpErr
		}
		return "", err
	}

	tx, err := s.db.Begin(ctx)
	if err != nil {
		return "", err
	}
	defer tx.Rollback(ctx)

	if err := ConsumeMFAChallenge(ctx, tx, challengeID, now); err != nil {
		return "", err
	}
	if _, err := tx.Exec(ctx, `
		update webauthn_credentials
		set sign_count = $3, last_used_at = $2
		where user_id = $1 and credential_id = $4
	`, challenge.UserID, now, cred.Authenticator.SignCount, cred.ID); err != nil {
		return "", err
	}

	token, err := s.issueSessionTx(ctx, tx, challenge.UserID, ip, ua, now, true /*reauth*/)
	if err != nil {
		return "", err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &challenge.UserID,
		Action:      "user.login_succeeded",
		EntityType:  "user",
		EntityID:    challenge.UserID.String(),
		At:          now,
	})
	if err := tx.Commit(ctx); err != nil {
		return "", err
	}
	return token, nil
}

// issueSessionTx inserts a session row within an existing tx. If reauthFresh
// is true, sessions.reauth_at is set to now (MFA ceremony counts as re-auth).
func (s *Service) issueSessionTx(ctx context.Context, tx pgx.Tx, userID uuid.UUID, ip, ua string, now interface{}, reauthFresh bool) (string, error) {
	// implementation shared with password login path — see sessions.go
	return s.insertSession(ctx, tx, userID, ip, ua, reauthFresh)
}
```

```go
// BeginPasskeyLogin starts a discoverable-credential assertion. We don't know
// which user yet — the authenticator will supply it via userHandle in the
// response. A challenge row is stored with user_id = nil-uuid for now; the
// real user_id is populated on completion. To respect the NOT NULL constraint,
// we write a sentinel user_id (see schema note) OR we store the initial state
// in a separate passkey_login_sessions table. We choose the latter below.
func (s *Service) BeginPasskeyLogin(ctx context.Context, ip, ua string) (uuid.UUID, *protocol.CredentialAssertion, error) {
	options, sessionData, err := s.webauthn.W.BeginDiscoverableLogin()
	if err != nil {
		return uuid.Nil, nil, err
	}
	id := uuid.Must(uuid.NewV7())
	stateJSON, err := json.Marshal(sessionData)
	if err != nil {
		return uuid.Nil, nil, err
	}
	// Reuse auth_mfa_challenges with user_id = nil-uuid sentinel? No — the
	// column is NOT NULL referencing users(id). Instead we persist the session
	// in a small passkey_login_sessions table added in Task 6.0.
	_, err = s.db.Exec(ctx, `
		insert into passkey_login_sessions (id, ip, user_agent, created_at, expires_at, webauthn_state)
		values ($1, $2, $3, $4, $5, $6)
	`, id, ip, ua, s.clock.Now(), s.clock.Now().Add(s.cfg.MFAChallengeTTL), stateJSON)
	if err != nil {
		return uuid.Nil, nil, err
	}
	return id, options, nil
}

// CompletePasskeyLogin finishes the discoverable ceremony, looks up the
// credential by credential_id, and issues a session.
func (s *Service) CompletePasskeyLogin(ctx context.Context, sessionID uuid.UUID, response []byte, ip, ua string) (string, error) {
	now := s.clock.Now()

	var stateJSON []byte
	var storedIP, storedUA string
	var expiresAt, consumedAt *time.Time
	err := s.db.QueryRow(ctx, `
		select webauthn_state, ip::text, user_agent, expires_at, consumed_at
		from passkey_login_sessions where id = $1
	`, sessionID).Scan(&stateJSON, &storedIP, &storedUA, &expiresAt, &consumedAt)
	if err != nil {
		return "", err
	}
	if consumedAt != nil {
		return "", ErrChallengeConsumed
	}
	if expiresAt != nil && now.After(*expiresAt) {
		return "", ErrChallengeExpired
	}
	if storedIP != ip || storedUA != ua {
		return "", ErrChallengeBinding
	}

	var sessionData webauthn.SessionData
	if err := json.Unmarshal(stateJSON, &sessionData); err != nil {
		return "", err
	}

	parsed, err := protocol.ParseCredentialRequestResponseBody(bytes.NewReader(response))
	if err != nil {
		return "", err
	}

	userHandler := func(rawID, userHandle []byte) (webauthn.User, error) {
		uid, err := uuid.FromBytes(userHandle)
		if err != nil {
			return nil, errors.New("bad user handle")
		}
		return s.loadWebAuthnUser(ctx, s.db, uid)
	}

	cred, err := s.webauthn.W.ValidateDiscoverableLogin(userHandler, sessionData, parsed)
	if err != nil {
		return "", err
	}

	// Recover user_id from webauthn_credentials row via credential_id.
	var userID uuid.UUID
	if err := s.db.QueryRow(ctx, `
		select user_id from webauthn_credentials where credential_id = $1
	`, cred.ID).Scan(&userID); err != nil {
		return "", err
	}

	tx, err := s.db.Begin(ctx)
	if err != nil {
		return "", err
	}
	defer tx.Rollback(ctx)

	if _, err := tx.Exec(ctx, `
		update passkey_login_sessions set consumed_at = $2 where id = $1
	`, sessionID, now); err != nil {
		return "", err
	}
	if _, err := tx.Exec(ctx, `
		update webauthn_credentials
		set sign_count = $3, last_used_at = $2
		where credential_id = $1
	`, cred.ID, now, cred.Authenticator.SignCount); err != nil {
		return "", err
	}
	token, err := s.insertSession(ctx, tx, userID, ip, ua, true /*reauth*/)
	if err != nil {
		return "", err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "user.login_succeeded",
		EntityType:  "user",
		EntityID:    userID.String(),
		At:          now,
	})
	return token, tx.Commit(ctx)
}
```

```go
// RequireFreshReauth is a middleware factory. A session qualifies as "fresh"
// if sessions.reauth_at is non-null and within `window`. Otherwise respond
// with 403 and body `{"code":"reauth_required"}` so the client opens the
// re-auth modal.
func RequireFreshReauth(window time.Duration) func(http.Handler) http.Handler {
	return func(next http.Handler) http.Handler {
		return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
			sess, ok := SessionFromContext(r.Context())
			if !ok {
				httpx.WriteError(w, http.StatusUnauthorized, "unauthenticated", "no session")
				return
			}
			now := ClockFromContext(r.Context()).Now()
			if sess.ReauthAt == nil || now.Sub(*sess.ReauthAt) > window {
				httpx.WriteErrorCode(w, http.StatusForbidden, "reauth_required", "step-up required")
				return
			}
			next.ServeHTTP(w, r)
		})
	}
}
```

```go
package auth

import (
	"context"
	"errors"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5"
)

// ReauthRequest carries one of three factors. Exactly one must be set.
type ReauthRequest struct {
	Password     string
	TOTPCode     string
	WebAuthnResp []byte // encoded CredentialAssertion response
}

// CompleteReauth verifies the supplied factor and, on success, bumps
// `sessions.reauth_at = now()` for the caller's session. Writes audit event.
func (s *Service) CompleteReauth(ctx context.Context, sessionID, userID uuid.UUID, req ReauthRequest) error {
	now := s.clock.Now()

	switch {
	case req.Password != "":
		user, err := s.GetUserByID(ctx, userID)
		if err != nil {
			return err
		}
		ok, err := argon2id.Verify(req.Password, user.PasswordHash)
		if err != nil {
			return err
		}
		if !ok {
			return ErrInvalidCredentials
		}
	case req.TOTPCode != "":
		if err := s.verifyTOTPCode(ctx, userID, req.TOTPCode); err != nil {
			return err
		}
	case len(req.WebAuthnResp) > 0:
		// Best-effort: validate against any of the user's credentials. This
		// path uses a *stored session* keyed by user; the caller must have
		// begun the ceremony via /auth/reauth/webauthn/options.
		if err := s.verifyReauthWebAuthn(ctx, userID, req.WebAuthnResp); err != nil {
			return err
		}
	default:
		return errors.New("no factor provided")
	}

	tx, err := s.db.Begin(ctx)
	if err != nil {
		return err
	}
	defer tx.Rollback(ctx)
	if err := bumpReauth(ctx, tx, sessionID, now); err != nil {
		return err
	}
	s.writeAudit(ctx, tx, auditEvent{
		ActorUserID: &userID,
		Action:      "mfa.reauth_completed",
		EntityType:  "session",
		EntityID:    sessionID.String(),
		At:          now,
	})
	return tx.Commit(ctx)
}

func bumpReauth(ctx context.Context, tx pgx.Tx, sessionID uuid.UUID, now time.Time) error {
	ct, err := tx.Exec(ctx, `
		update sessions set reauth_at = $2 where id = $1
	`, sessionID, now)
	if err != nil {
		return err
	}
	if ct.RowsAffected() == 0 {
		return errors.New("session not found")
	}
	return nil
}
```

**`web/lib/hooks/use-reauth.ts`:**

```tsx
"use client";

import { useState } from "react";

type ReauthPending = {
  resolve: (v: unknown) => void;
  reject: (e: unknown) => void;
  retry: () => Promise<unknown>;
};

export function useReauth() {
  const [reauthState, setReauthState] = useState<ReauthPending | null>(null);

  async function withReauth<T>(fn: () => Promise<T>): Promise<T> {
    try {
      return await fn();
    } catch (e) {
      const err = e as { code?: string };
      if (err?.code === "reauth_required") {
        return await new Promise<T>((resolve, reject) => {
          setReauthState({
            resolve: (v) => resolve(v as T),
            reject,
            retry: fn as () => Promise<unknown>,
          });
        });
      }
      throw e;
    }
  }

  return { withReauth, reauthState, setReauthState };
}
```

```tsx
import { apiFetch } from "@/lib/api/client";

type ReauthState = {
  resolve: (v: unknown) => void;
  reject: (e: unknown) => void;
  retry: () => Promise<unknown>;
} | null;

export function ReauthModal({ state, onClose }: { state: ReauthState; onClose: () => void }) {
  const [password, setPassword] = useState("");
  const [totp, setTotp] = useState("");
  const [err, setErr] = useState<string | null>(null);
```
